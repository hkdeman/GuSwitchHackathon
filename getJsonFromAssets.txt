String json = null;
        try {
            InputStream is = getAssets().open("vocabulary.json");
            int size = is.available();
            byte[] buffer = new byte[size];
            is.read(buffer);
            is.close();
            json = new String(buffer, "UTF-8");
        } catch (IOException ex) {
            ex.printStackTrace();
        }

        try {
            vocabulary = new JSONArray(json);
        } catch (JSONException e) {
            e.printStackTrace();
            vocabulary = null;
        }
